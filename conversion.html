import tkinter as tk from tkinter import ttk class ConversionApp: def __init__(self, root): self.root = root self.root.title("Physics Conversion App for Kids") self.conversion_types = { "Length": ["Meters", "Kilometers", "Centimeters", "Millimeters"], "Mass": ["Grams", "Kilograms", "Milligrams"], "Time": ["Seconds", "Minutes", "Hours"], "Temperature": ["Celsius", "Fahrenheit", "Kelvin"] } self.create_widgets() def create_widgets(self): self.conversion_type_label = ttk.Label(self.root, text="Select Conversion Type:") self.conversion_type_label.grid(row=0, column=0, padx=10, pady=10) self.conversion_type = ttk.Combobox(self.root, values=list(self.conversion_types.keys())) self.conversion_type.grid(row=0, column=1, padx=10, pady=10) self.conversion_type.bind("<>", self.update_units) self.from_unit_label = ttk.Label(self.root, text="From:") self.from_unit_label.grid(row=1, column=0, padx=10, pady=10) self.from_unit = ttk.Combobox(self.root) self.from_unit.grid(row=1, column=1, padx=10, pady=10) self.to_unit_label = ttk.Label(self.root, text="To:") self.to_unit_label.grid(row=2, column=0, padx=10, pady=10) self.to_unit = ttk.Combobox(self.root) self.to_unit.grid(row=2, column=1, padx=10, pady=10) self.value_label = ttk.Label(self.root, text="Value:") self.value_label.grid(row=3, column=0, padx=10, pady=10) self.value_entry = ttk.Entry(self.root) self.value_entry.grid(row=3, column=1, padx=10, pady=10) self.convert_button = ttk.Button(self.root, text="Convert", command=self.convert) self.convert_button.grid(row=4, column=0, columnspan=2, padx=10, pady=10) self.result_label = ttk.Label(self.root, text="Result:") self.result_label.grid(row=5, column=0, padx=10, pady=10) self.result = ttk.Label(self.root, text="") self.result.grid(row=5, column=1, padx=10, pady=10) def update_units(self, event): conversion_type = self.conversion_type.get() units = self.conversion_types.get(conversion_type, []) self.from_unit['values'] = units self.to_unit['values'] = units def convert(self): conversion_type = self.conversion_type.get() from_unit = self.from_unit.get() to_unit = self.to_unit.get() value = float(self.value_entry.get()) if conversion_type == "Length": result = self.convert_length(from_unit, to_unit, value) elif conversion_type == "Mass": result = self.convert_mass(from_unit, to_unit, value) elif conversion_type == "Time": result = self.convert_time(from_unit, to_unit, value) elif conversion_type == "Temperature": result = self.convert_temperature(from_unit, to_unit, value) self.result.config(text=f"{result:.2f} {to_unit}") def convert_length(self, from_unit, to_unit, value): conversions = { "Meters": 1, "Kilometers": 1000, "Centimeters": 0.01, "Millimeters": 0.001 } return value * conversions[from_unit] / conversions[to_unit] def convert_mass(self, from_unit, to_unit, value): conversions = { "Grams": 1, "Kilograms": 1000, "Milligrams": 0.001 } return value * conversions[from_unit] / conversions[to_unit] def convert_time(self, from_unit, to_unit, value): conversions = { "Seconds": 1, "Minutes": 60, "Hours": 3600 } return value * conversions[from_unit] / conversions[to_unit] def convert_temperature(self, from_unit, to_unit, value): if from_unit == "Celsius": if to_unit == "Fahrenheit": return (value * 9/5) + 32 elif to_unit == "Kelvin": return value + 273.15 elif from_unit == "Fahrenheit": if to_unit == "Celsius": return (value - 32) * 5/9 elif to_unit == "Kelvin": return (value - 32) * 5/9 + 273.15 elif from_unit == "Kelvin": if to_unit == "Celsius": return value - 273.15 elif to_unit == "Fahrenheit": return (value - 273.15) * 9/5 + 32 return value if __name__ == "__main__": root = tk.Tk() app = ConversionApp(root) root.mainloop()
